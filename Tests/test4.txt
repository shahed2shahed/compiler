import { Component } from '@angular/core';
import { bootstrapApplication } from '@angular/platform-browser';

@Component({
  selector: 'app-root',
  template: `
    <div>
      <button (click)="countingSum(counter)" >Increase</button>
      <button (click)="countingDic(counter)" >Decrease</button>

      <p>
        {{ counter > 0
          ? 'The number is positive: ' + counter
          : counter < 0
          ? 'The number is negative: ' + counter
          : 'The number is zero' }}
      </p>
    </div>
  `,
})
export class AppComponent {
  counter = 0 ;
  countingSum(index: number): void {
    this.counter = index + 1;
  }

  countingDic(index: number): void {
    this.counter = index - 1;
  }
}

 bootstrapApplication(AppComponent)
   .catch((err: any) => console.error(err));
