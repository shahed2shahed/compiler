import { Component } from '@angular/core';

@Component({
  selector: 'app-root',
    selector: 'app-root',
  standalone: true,
  template: `

      </button>

  `
}),

export class AppComponent {
  name: string = 'Angular';

}

bootstrapApplication(AppComponent).catch((err: any) => console.error(err));
